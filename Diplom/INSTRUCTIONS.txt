  - Шрифт: Consolas или Courier New, 10pt
  - Выравнивание: по левому краю
  - Фон: светло-серый (по желанию)
ФРАГМЕНТ КОДА (из Assets/Script/Inventory/CraftingManager.cs):
```csharp
private List<Item[,]> NormalizeRecipe(Item[,] recipe)
{
    // Найти bounding box заполненных ячеек
    int minRow = 3, maxRow = 0, minCol = 3, maxCol = 0;
    for (int r = 0; r < 3; r++)
        for (int c = 0; c < 3; c++)
            if (recipe[r, c] != null)
            {
                minRow = Mathf.Min(minRow, r);
                maxRow = Mathf.Max(maxRow, r);
                minCol = Mathf.Min(minCol, c);
                maxCol = Mathf.Max(maxCol, c);
            }
    // Смещение к левому верхнему углу
    Item[,] normalized = new Item[maxRow - minRow + 1, maxCol - minCol + 1];
    for (int r = minRow; r <= maxRow; r++)
        for (int c = minCol; c <= maxCol; c++)
            normalized[r - minRow, c - minCol] = recipe[r, c];
    return normalized;
}
```

private void Update()
{
    switch (currentState)
    {
        case AnimalState.Idle:
            idleTimer -= Time.deltaTime;
            if (idleTimer <= 0)
            {
                SetState(AnimalState.Roaming);
            }
            break;
        case AnimalState.Roaming:
            if (!agent.pathPending && agent.remainingDistance < 0.1f)
            {
                SetState(AnimalState.Idle);
            }
            break;
    }
}

private void SetState(AnimalState newState)
{
    currentState = newState;
    if (newState == AnimalState.Roaming)
    {
        Vector2 randomPoint = initialPosition + Random.insideUnitCircle * roamRadius;
        agent.SetDestination(randomPoint);
    }
    else
    {
        agent.SetDestination(transform.position);
        idleTimer = Random.Range(2f, 5f);
    }
}
```

Листинг 5 - Система времени суток

ЧТО ВСТАВИТЬ: Код из GlobalTime.cs - расчёт времени и фаз

private void Update()
{
    timeCounter += Time.deltaTime;
    if (timeCounter >= 1f) // 1 реальная секунда = 1 игровая минута
    {
        timeCounter -= 1f;
        minutes++;
        if (minutes >= 60)
        {
            minutes = 0;
            hours++;
            if (hours >= 24)
            {
                hours = 0;
                dayNumber++;
            }
        }
        UpdateTimePhase();
    }
}

private void UpdateTimePhase()
{
    if (hours >= 6 && hours < 12) currentPhase = TimePhase.Morning;
    else if (hours >= 12 && hours < 18) currentPhase = TimePhase.Day;
    else if (hours >= 18 && hours < 22) currentPhase = TimePhase.Evening;
    else currentPhase = TimePhase.Night;
}
```

6. ПОСЛЕДОВАТЕЛЬНОСТЬ ДЕЙСТВИЙ ДЛЯ ЗАВЕРШЕНИЯ
========================================================================

ШАГ 1: Сделать скриншоты (25 штук)
  - Запустить Unity
  - Открыть проект GameYear
  - Запустить игру (Play)
  - Делать скриншоты через Win+Shift+S или PrintScreen
  - Сохранить в папку: Diplom\Screenshots\

ШАГ 2: Создать диаграммы (5 штук)
  - Открыть https://draw.io
  - Создать каждую диаграмму по описанию выше
  - Экспортировать в PNG
  - Сохранить в папку: Diplom\Diagrams\

ШАГ 3: Вставить в документ
  - Открыть Diplom_полный.docx
  - Перейти к каждому месту по инструкции
  - Вставка → Рисунки → выбрать файл
  - Под каждым рисунком добавить подпись:
    "Рисунок X — Название"
  - Выравнивание: по центру
  - Шрифт подписи: Times New Roman, 12pt

ШАГ 4: Вставить листинги кода
  - В Word: Вставка → Таблица 1x1
  - Вставить код из инструкции
  - Шрифт: Consolas, 10pt
  - Под таблицей добавить подпись:
    "Листинг X — Название"

ШАГ 5: Проверить нумерацию
  - Все рисунки должны быть пронумерованы по порядку
  - Все таблицы должны быть пронумерованы по порядку
  - Все листинги должны быть пронумерованы по порядку
  - Ссылки в тексте должны соответствовать номерам

ШАГ 6: Добавить содержание
  - Ссылки → Оглавление → Автособираемое оглавление
  - Обновить поле перед печатью
